    <script type="text/babel">
        const { createContext, useContext, useState, useEffect } = React;
        const { BrowserRouter: Router, Routes, Route, Navigate, Link, useNavigate } = ReactRouterDOM;

        // ============= AUTH CONTEXT =============
        const AuthContext = createContext();

        const useAuth = () => {
            const context = useContext(AuthContext);
            if (!context) throw new Error('useAuth must be used within AuthProvider');
            return context;
        };

        const AuthProvider = ({ children }) => {
            const [user, setUser] = useState(null);
            const [loading, setLoading] = useState(true);

            useEffect(() => {
                const savedUser = localStorage.getItem('user');
                if (savedUser) {
                    setUser(JSON.parse(savedUser));
                }
                setLoading(false);
            }, []);

            const login = async (email, password) => {
                try {
                    const userData = { name: 'John Doe', email, id: '123' };
                    setUser(userData);
                    localStorage.setItem('user', JSON.stringify(userData));
                    return { success: true };
                } catch (error) {
                    return { success: false, message: 'Login failed' };
                }
            };

            const register = async (name, email, password, phone) => {
                try {
                    const userData = { name, email, phone, id: '123' };
                    setUser(userData);
                    localStorage.setItem('user', JSON.stringify(userData));
                    return { success: true };
                } catch (error) {
                    return { success: false, message: 'Registration failed' };
                }
            };

            const logout = () => {
                setUser(null);
                localStorage.removeItem('user');
            };

            return React.createElement(AuthContext.Provider, {
                value: { user, loading, login, register, logout }
            }, children);
        };

        // ============= NAVBAR =============
        const Navbar = () => {
            const { user, logout } = useAuth();

            return React.createElement('nav', null,
                React.createElement('div', { className: 'nav-container' },
                    React.createElement('div', { className: 'nav-links' },
                        React.createElement(Link, { to: "/dashboard", style: { fontSize: '20px', fontWeight: 'bold', color: '#2563eb' } }, "WealthMind"),
                        React.createElement(Link, { to: "/dashboard" }, "Dashboard"),
                        React.createElement(Link, { to: "/market" }, "Market"),
                        React.createElement(Link, { to: "/portfolio" }, "Portfolio")
                    ),
                    React.createElement('div', { className: 'nav-links' },
                        React.createElement(Link, { to: "/profile" }, user?.name),
                        React.createElement('button', { 
                            onClick: logout, 
                            className: 'btn-secondary',
                            style: { marginLeft: '16px' }
                        }, "Logout")
                    )
                )
            );
        };

        // ============= LOGIN PAGE =============
        const Login = () => {
            const [email, setEmail] = useState('');
            const [password, setPassword] = useState('');
            const [error, setError] = useState('');
            const { login } = useAuth();
            const navigate = useNavigate();

            const handleSubmit = async (e) => {
                e.preventDefault();
                const result = await login(email, password);
                if (result.success) {
                    navigate('/dashboard');
                } else {
                    setError(result.message);
                }
            };

            return React.createElement('div', { className: 'min-h-screen flex items-center justify-center bg-gradient' },
                React.createElement('div', { className: 'max-w-md w-full' },
                    React.createElement('div', { className: 'text-center mb-8' },
                        React.createElement('div', { 
                            style: { 
                                display: 'inline-block',
                                backgroundColor: '#2563eb', 
                                color: 'white',
                                width: '64px',
                                height: '64px',
                                borderRadius: '16px',
                                display: 'flex',
                                alignItems: 'center',
                                justifyContent: 'center',
                                fontWeight: 'bold',
                                fontSize: '24px',
                                marginBottom: '16px'
                            } 
                        }, "W"),
                        React.createElement('h1', { style: { fontSize: '36px', fontWeight: 'bold', color: '#1f2937', marginBottom: '8px' } }, "WealthMind"),
                        React.createElement('p', { style: { color: '#6b7280' } }, "Smart Investment Platform")
                    ),
                    React.createElement('div', { className: 'card' },
                        React.createElement('h2', { style: { fontSize: '24px', fontWeight: 'bold', color: '#1f2937', marginBottom: '24px' } }, "Login to your account"),
                        error && React.createElement('div', { 
                            className: 'bg-red-50 border-red-200 text-red-600 p-4 rounded-lg mb-4' 
                        }, error),
                        React.createElement('form', { onSubmit: handleSubmit, className: 'space-y-4' },
                            React.createElement('div', null,
                                React.createElement('label', { className: 'label' }, "Email"),
                                React.createElement('input', { 
                                    type: 'email', 
                                    className: 'input-field', 
                                    placeholder: 'john@example.com',
                                    value: email, 
                                    onChange: (e) => setEmail(e.target.value),
                                    required: true
                                })
                            ),
                            React.createElement('div', null,
                                React.createElement('label', { className: 'label' }, "Password"),
                                React.createElement('input', { 
                                    type: 'password', 
                                    className: 'input-field', 
                                    placeholder: 'Enter your password',
                                    value: password, 
                                    onChange: (e) => setPassword(e.target.value),
                                    required: true
                                })
                            ),
                            React.createElement('button', { type: 'submit', className: 'btn-primary' }, "Login")
                        ),
                        React.createElement('div', { className: 'text-center mt-6' },
                            React.createElement('p', { style: { color: '#6b7280' } }, 
                                "Don't have an account? ",
                                React.createElement(Link, { to: "/register", style: { color: '#2563eb', fontWeight: '600' } }, "Register")
                            )
                        )
                    )
                )
            );
        };

        // ============= REGISTER PAGE =============
        const Register = () => {
            const [formData, setFormData] = useState({ name: '', email: '', password: '', confirmPassword: '', phone: '' });
            const [error, setError] = useState('');
            const { register } = useAuth();
            const navigate = useNavigate();

            const handleChange = (e) => {
                setFormData({ ...formData, [e.target.name]: e.target.value });
            };

            const handleSubmit = async (e) => {
                e.preventDefault();
                if (formData.password !== formData.confirmPassword) {
                    setError('Passwords do not match');
                    return;
                }
                const result = await register(formData.name, formData.email, formData.password, formData.phone);
                if (result.success) {
                    navigate('/dashboard');
                } else {
                    setError(result.message);
                }
            };

            return React.createElement('div', { className: 'min-h-screen flex items-center justify-center bg-gradient' },
                React.createElement('div', { className: 'max-w-md w-full' },
                    React.createElement('div', { className: 'text-center mb-8' },
                        React.createElement('div', { 
                            style: { 
                                display: 'inline-block',
                                backgroundColor: '#2563eb', 
                                color: 'white',
                                width: '64px',
                                height: '64px',
                                borderRadius: '16px',
                                display: 'flex',
                                alignItems: 'center',
                                justifyContent: 'center',
                                fontWeight: 'bold',
                                fontSize: '24px',
                                marginBottom: '16px'
                            } 
                        }, "W"),
                        React.createElement('h1', { style: { fontSize: '36px', fontWeight: 'bold', color: '#1f2937', marginBottom: '8px' } }, "WealthMind"),
                        React.createElement('p', { style: { color: '#6b7280' } }, "Create your investment account")
                    ),
                    React.createElement('div', { className: 'card' },
                        React.createElement('h2', { style: { fontSize: '24px', fontWeight: 'bold', color: '#1f2937', marginBottom: '24px' } }, "Register"),
                        error && React.createElement('div', { 
                            className: 'bg-red-50 border-red-200 text-red-600 p-4 rounded-lg mb-4' 
                        }, error),
                        React.createElement('form', { onSubmit: handleSubmit, className: 'space-y-4' },
                            React.createElement('div', null,
                                React.createElement('label', { className: 'label' }, "Full Name"),
                                React.createElement('input', { 
                                    type: 'text',
                                    name: 'name',
                                    className: 'input-field', 
                                    placeholder: 'John Doe',
                                    value: formData.name, 
                                    onChange: handleChange,
                                    required: true
                                })
                            ),
                            React.createElement('div', null,
                                React.createElement('label', { className: 'label' }, "Email"),
                                React.createElement('input', { 
                                    type: 'email',
                                    name: 'email',
                                    className: 'input-field', 
                                    placeholder: 'john@example.com',
                                    value: formData.email, 
                                    onChange: handleChange,
                                    required: true
                                })
                            ),
                            React.createElement('div', null,
                                React.createElement('label', { className: 'label' }, "Phone (Optional)"),
                                React.createElement('input', { 
                                    type: 'tel',
                                    name: 'phone',
                                    className: 'input-field', 
                                    placeholder: '+91-9876543210',
                                    value: formData.phone, 
                                    onChange: handleChange
                                })
                            ),
                            React.createElement('div', null,
                                React.createElement('label', { className: 'label' }, "Password"),
                                React.createElement('input', { 
                                    type: 'password',
                                    name: 'password',
                                    className: 'input-field', 
                                    placeholder: 'At least 6 characters',
                                    value: formData.password, 
                                    onChange: handleChange,
                                    required: true
                                })
                            ),
                            React.createElement('div', null,
                                React.createElement('label', { className: 'label' }, "Confirm Password"),
                                React.createElement('input', { 
                                    type: 'password',
                                    name: 'confirmPassword',
                                    className: 'input-field', 
                                    placeholder: 'Re-enter your password',
                                    value: formData.confirmPassword, 
                                    onChange: handleChange,
                                    required: true
                                })
                            ),
                            React.createElement('button', { type: 'submit', className: 'btn-primary' }, "Register")
                        ),
                        React.createElement('div', { className: 'text-center mt-6' },
                            React.createElement('p', { style: { color: '#6b7280' } }, 
                                "Already have an account? ",
                                React.createElement(Link, { to: "/login", style: { color: '#2563eb', fontWeight: '600' } }, "Login")
                            )
                        )
                    )
                )
            );
        };

        // ============= DASHBOARD =============
        const Dashboard = () => {
            const { user } = useAuth();

            return React.createElement('div', { className: 'container' },
                React.createElement('div', { className: 'mb-8' },
                    React.createElement('h1', { style: { fontSize: '30px', fontWeight: 'bold', color: '#1f2937' } }, `Welcome back, ${user?.name}! ðŸ‘‹`),
                    React.createElement('p', { style: { color: '#6b7280', marginTop: '8px' } }, "Here's your investment overview")
                ),
                React.createElement('div', { className: 'grid grid-cols-3 mb-8' },
                    React.createElement('div', { 
                        className: 'card',
                        style: { background: 'linear-gradient(135deg, #3b82f6 0%, #2563eb 100%)', color: 'white' }
                    },
                        React.createElement('h3', { style: { fontSize: '18px', fontWeight: '600', opacity: 0.9, marginBottom: '16px' } }, "Total Invested"),
                        React.createElement('p', { style: { fontSize: '30px', fontWeight: 'bold' } }, "â‚¹0")
                    ),
                    React.createElement('div', { 
                        className: 'card',
                        style: { background: 'linear-gradient(135deg, #10b981 0%, #059669 100%)', color: 'white' }
                    },
                        React.createElement('h3', { style: { fontSize: '18px', fontWeight: '600', opacity: 0.9, marginBottom: '16px' } }, "Current Value"),
                        React.createElement('p', { style: { fontSize: '30px', fontWeight: 'bold' } }, "â‚¹0")
                    ),
                    React.createElement('div', { 
                        className: 'card',
                        style: { background: 'linear-gradient(135deg, #8b5cf6 0%, #7c3aed 100%)', color: 'white' }
                    },
                        React.createElement('h3', { style: { fontSize: '18px', fontWeight: '600', opacity: 0.9, marginBottom: '16px' } }, "Total P&L"),
                        React.createElement('p', { style: { fontSize: '30px', fontWeight: 'bold' } }, "â‚¹0")
                    )
                ),
                React.createElement('div', { className: 'card' },
                    React.createElement('h2', { style: { fontSize: '20px', fontWeight: 'bold', color: '#1f2937', marginBottom: '16px' } }, "Getting Started"),
                    React.createElement('p', { style: { color: '#6b7280', marginBottom: '16px' } }, "Your dashboard is ready! Connect your backend API to see live data."),
                    React.createElement('div', { className: 'flex gap-4' },
                        React.createElement(Link, { to: "/market", className: 'btn-primary', style: { width: 'auto', display: 'inline-block' } }, "Explore Market"),
                        React.createElement(Link, { to: "/profile", className: 'btn-secondary' }, "Complete Profile")
                    )
                )
            );
        };

        // ============= MARKET =============
        const Market = () => {
            return React.createElement('div', { className: 'container' },
                React.createElement('h1', { style: { fontSize: '30px', fontWeight: 'bold', color: '#1f2937', marginBottom: '24px' } }, "Stock Market"),
                React.createElement('div', { className: 'card' },
                    React.createElement('p', { style: { color: '#6b7280' } }, "Connect your backend API to display stocks here.")
                )
            );
        };

        // ============= PORTFOLIO =============
        const Portfolio = () => {
            return React.createElement('div', { className: 'container' },
                React.createElement('h1', { style: { fontSize: '30px', fontWeight: 'bold', color: '#1f2937', marginBottom: '24px' } }, "My Portfolio"),
                React.createElement('div', { className: 'card' },
                    React.createElement('p', { style: { color: '#6b7280' } }, "Your portfolio will appear here once you make investments.")
                )
            );
        };

        // ============= PROFILE =============
        const Profile = () => {
            const { user } = useAuth();

            return React.createElement('div', { className: 'container' },
                React.createElement('h1', { style: { fontSize: '30px', fontWeight: 'bold', color: '#1f2937', marginBottom: '24px' } }, "Profile Settings"),
                React.createElement('div', { className: 'card' },
                    React.createElement('div', { className: 'text-center mb-6' },
                        React.createElement('div', { 
                            style: { 
                                display: 'inline-block',
                                backgroundColor: '#2563eb',
                                color: 'white',
                                width: '80px',
                                height: '80px',
                                borderRadius: '50%',
                                display: 'flex',
                                alignItems: 'center',
                                justifyContent: 'center',
                                fontWeight: 'bold',
                                fontSize: '30px',
                                marginBottom: '16px'
                            }
                        }, user?.name?.charAt(0).toUpperCase()),
                        React.createElement('h2', { style: { fontSize: '20px', fontWeight: 'bold', color: '#1f2937' } }, user?.name),
                        React.createElement('p', { style: { color: '#6b7280', marginTop: '4px' } }, user?.email)
                    )
                )
            );
        };

        // ============= PROTECTED ROUTE =============
        const ProtectedRoute = ({ children }) => {
            const { user, loading } = useAuth();
            
            if (loading) {
                return React.createElement('div', { className: 'min-h-screen flex items-center justify-center' },
                    React.createElement('div', { 
                        className: 'animate-spin',
                        style: { 
                            borderRadius: '50%',
                            height: '48px',
                            width: '48px',
                            borderBottom: '2px solid #2563eb'
                        }
                    })
                );
            }
            
            return user ? children : React.createElement(Navigate, { to: "/login", replace: true });
        };

        const PublicRoute = ({ children }) => {
            const { user, loading } = useAuth();
            
            if (loading) {
                return React.createElement('div', { className: 'min-h-screen flex items-center justify-center' },
                    React.createElement('div', { 
                        className: 'animate-spin',
                        style: { 
                            borderRadius: '50%',
                            height: '48px',
                            width: '48px',
                            borderBottom: '2px solid #2563eb'
                        }
                    })
                );
            }
            
            return user ? React.createElement(Navigate, { to: "/dashboard", replace: true }) : children;
        };

        // ============= MAIN APP =============
        const App = () => {
            const { user } = useAuth();

            return React.createElement(Router, null,
                React.createElement('div', { className: 'min-h-screen', style: { backgroundColor: '#f9fafb' } },
                    user && React.createElement(Navbar),
                    React.createElement(Routes, null,
                        React.createElement(Route, { path: "/login", element: React.createElement(PublicRoute, null, React.createElement(Login)) }),
                        React.createElement(Route, { path: "/register", element: React.createElement(PublicRoute, null, React.createElement(Register)) }),
                        React.createElement(Route, { path: "/dashboard", element: React.createElement(ProtectedRoute, null, React.createElement(Dashboard)) }),
                        React.createElement(Route, { path: "/market", element: React.createElement(ProtectedRoute, null, React.createElement(Market)) }),
                        React.createElement(Route, { path: "/portfolio", element: React.createElement(ProtectedRoute, null, React.createElement(Portfolio)) }),
                        React.createElement(Route, { path: "/profile", element: React.createElement(ProtectedRoute, null, React.createElement(Profile)) }),
                        React.createElement(Route, { path: "/", element: React.createElement(Navigate, { to: "/dashboard", replace: true }) })
                    )
                )
            );
        };

        // ============= RENDER =============
        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(
            React.createElement(React.StrictMode, null,
                React.createElement(AuthProvider, null,
                    React.createElement(App)
                )
            )
        );
    </script>
